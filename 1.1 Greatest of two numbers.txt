       num3 = scan.nextDouble();

     
       if(num1 >= num2) 
	{
            if(num1 >= num3) 
		largestNumber = num1;
            else 
		largestNumber = num3;
       } else 
	{
            if(num2>=num3) 
                largestNumber = num2;
            else largestNumber = num3;
       }

       System.out.println("Largest number = "+ largestNumber);
    

     }
 }
